		function binarySearchNonRecursive (arrayA, N, X) {

		'use strict';

		var low = 0,
		    high = N - 1,
		    mid;

		    while (low < high) 
		    {
		    	var mid = low + (high - low)/2;  		// (low + high) may overflow if both numbrs are near max int for the system

		    	if (X === arrayA[mid]) {
		    		return mid;				// Found X
		    	}

		    	if (X < arrayA[mid]) {
		    		high = --mid;				// X lies before mid
		    	} else {
		    		low = ++mid;				// X lies after mid
		    	}

		    }

		    return false;
		}

	function primeFactorizationCheck ( num ) {

		'use strict';

		// check edge cases 
		if ( isNaN(num) ) {
			console.log(num + ' is NOT A NUMBER/VALUE');
			return false;
		}

		if ( !isFinite(num) ) {
			console.log(num + ' is NOT a FINITE legal number on this system');
			return false;
		}

		if ( (num < 1) || (num === 1) ) {
			console.log(num + ' is NOT PRIME');
			return false;
		}

		if ( (num % 1 !== 0 ) ) {
			console.log(num + ' is NOT an integer');
			return false;
		}

		// Check for all Prime Factors of num

		console.log('Check for all PRIME FACTORS of  ' + num );

		//var primeRange = Math.sqrt(num);   // num gets refactored, so this won't save us any time
		var count;

		for (var i=2; i <= Math.sqrt(num); i++ ) {
			if ( num % i === 0 ) {
				console.log( i + ' is a factor of ' + num + '...now check frequency');

				count = 0;

					while ( num %  i  ===  0 ) {
						num = num / i;
						count++;
					}

				console.log(i + ' , ' + count);	

			}
		}

		if ( num !== 1 ) {
			console.log( num + ' , 1');

		}

		return true;

	}
